def premium_emoji(name: str, fallback: str = "") -> str:
    emoji_id = custom_emoji_id(name)
    if not emoji_id or emoji_id == "1234567890123456789":
        return fallback
    return f'<tg-emoji emoji-id="{emoji_id}">{fallback}</tg-emoji>'


def em(name: str, fallback: str = "") -> str:
    return premium_emoji(name, fallback)


def line(name: str, text: str, fallback: str = "•") -> str:
    return TEXT_LINE_TEMPLATE.format(emoji=em(name, fallback), text=text)


def line_key(name: str, index: int) -> str:
    return name if index <= 0 else f"{name}_{index}"


def indexed_line(name: str, text: str, index: int = 0, fallback: str = "•") -> str:
    return line(line_key(name, index), text, fallback)


def emojify_lines(name: str, text: str, fallback: str = "•") -> str:
    lines = []
    non_empty_index = 0
    for raw_line in text.splitlines():
        if raw_line.strip():
            lines.append(indexed_line(name, raw_line, non_empty_index, fallback))
            non_empty_index += 1
        else:
            lines.append("")
    return "\n".join(lines)


def custom_emoji_id(name: str) -> str | None:
    emoji_id = CUSTOM_EMOJI.get(name)
    if not emoji_id or emoji_id == "1234567890123456789":
        return None
    return emoji_id
